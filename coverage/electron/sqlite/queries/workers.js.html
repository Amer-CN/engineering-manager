                totalWage: Math.round(totalWage * 100) / 100,
            });
        }
        return {
            teamId: teamId,
            teamName: teamName,
            workerCount: pwRows.length,
            teamTotal: Math.round(teamTotal * 100) / 100,
            details: details,
        };
    }
    catch (err) {
        log.error('[SQLite] workers.getTeamWages error:', err);
        return null;
    }
}
